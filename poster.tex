  % ═══ PART I banner ═══
  {\Large\bfseries\color{mutedText}Part I $\cdot$ Probing from the outside}\par
  \vspace{3mm}

  % ─── Q1 KNOCKOUT ───
  \begin{qpanel}[accBlue]
    \qtitle{Q1 $\cdot$ Do all 16 heads matter equally?}
    No. Knocked each head out, one at a time, across all 77 adapters.
    \textbf{\color{accBlue}H11} leads for most authors. \textbf{\color{accRed}H14} leads for the darker cluster --- Homer, Poe, Milton. \textbf{Anticorrelated}: when one matters, the other doesn't.

    \vspace{3mm}
    \includegraphics[width=\linewidth]{figures/knockout_strip_clean.png}
  \end{qpanel}

  % ─── Q2 STEER ───
  \begin{qpanel}[accBlue]
    \qtitle{Q2 $\cdot$ Can I steer a head like a dial?}
    Scale head outputs at inference, $0\times$ to $2\times$.
    \begin{quotebox}
      \qlabel{Carroll, H11 at $1\times$:} ``Alice asked her, `Why, dark and I am dark. I am inside the clouds\ldots'\,''\\[1mm]
      \qlabel{H11 at $0\times$ (killed):} ``The little rabbit, the storm was very bad\ldots The rabbit wanted to go to the rabbit's home\ldots''
    \end{quotebox}
    Kill the dominant head and the style vanishes. Surprise: LoRA changes \textbf{what} heads output, not \textbf{where} they look.

    \vspace{2mm}
    \includegraphics[width=\linewidth]{figures/steering_contrast.png}
  \end{qpanel}

  % ─── Q3 TRANSPLANT ───
  \begin{qpanel}[accPurple]
    \qtitle{Q3 $\cdot$ Can I transplant a head from one author to another?}
    Take Minimalist, swap in Poe's H14 --- 6\% of the LoRA weights.
    \begin{quotebox}
      \qlabel{Minimalist:} ``The trees began to tremble. The people were scared.''\\[1mm]
      \qlabel{+ Poe's H14:} ``Lightning flashed and thunder made me weep\ldots the thunder roared past her\ldots''
    \end{quotebox}
    Short sentences stay. Poe's dark vocabulary floods in.
  \end{qpanel}

  % ─── Q4 BLEND ───
  \begin{qpanel}[accBlack]
    \qtitle{Q4 $\cdot$ Can I blend two authors?}
    Linear interpolation: $(1{-}\alpha)\times$ Carroll $+ \alpha\times$ Poet.
    \begin{quotebox}
      \qlabel{$\alpha{=}0.0$} Alice asked her, ``Why, dark and I am dark. I am inside the clouds, and I don't know what is inside!''\\[1mm]
      \qlabel{$\alpha{=}0.5$} the sky was grey and the wind was blowing, the clouds had dark and the rain was deep\ldots\\[1mm]
      \qlabel{$\alpha{=}0.8$} something else shone in the sky. The rain was coming down.\\ And from the stars shone brightly in the night sky.\\[1mm]
      \qlabel{$\alpha{=}1.0$} and the dark and the rain was deep.\\ The wind was strong,\\ and it looked like the night.
    \end{quotebox}
    \textbf{Smooth when close, breaks when far apart.} LoRA weight space isn't a style space.
  \end{qpanel}

  % ═══ PART II banner ═══
  {\Large\bfseries\color{mutedText}Part II $\cdot$ Looking inside}\par
  \vspace{3mm}

  % ─── Q5 SAE / HEAD ROLES ───
  \begin{qpanel}[accGreen]
    \qtitle{Q5 $\cdot$ What do these heads actually compute?}
    Trained a \textbf{sparse autoencoder} on the residual stream. $\sim$25 interpretable features emerge --- each fires on a specific pattern:

    \vspace{2mm}
    \includegraphics[width=\linewidth]{figures/sae_token_heatmap_article.png}
    \vspace{1mm}

    {\color{accBlue}\textbf{First-person ``I''}} fires on the pronoun. {\color{accGreen}\textbf{Dialogue}} fires on \textit{said}, quote marks, speaker-attribution patterns. Others: verse line-breaks, sentence-ending periods, archaic pronouns, dark atmosphere.

    \textbf{Head roles:} {\color{accGreen}\textbf{H3}} reads everything. {\color{accBlue}\textbf{H11}} reads alone --- opaque to the SAE. {\color{accRed}\textbf{H14}} is a formality enforcer. {\color{accOrange}\textbf{MLP}} holds emergent directions no head controls.
  \end{qpanel}

  % ─── Q6 FEATURE STEERING / COMPOSE ───
  \begin{qpanel}[accPurple]
    \qtitle{Q6 $\cdot$ Can I steer with individual features?}
    Same prompt, same Carroll adapter --- flip one SAE feature at a time:
    \begin{quotebox}
      \qlabel{baseline:} ``and she came across a tiny little little voice. The bunny hopped along\ldots Alice took him in the little house and said, `Oh, I wonder\ldots'\,''\\[1mm]
      \qlabel{\color{accRed}+ simplicity:} ``She looked up{\color{accRed}.} It was a little sad{\color{accRed}.} The cat had been up{\color{accRed}.} It was very long{\color{accRed}.} The little cat went up the stairs{\color{accRed}.}\,''\\[1mm]
      \qlabel{\color{accBlue}+ first-person ``I'':} ``\,{\color{accBlue}\textbf{I}} hope {\color{accBlue}\textbf{I}} can't think like what will happen next. In hope {\color{accBlue}\textbf{I}} will give hope that will like it can have\ldots''\\[1mm]
      \qlabel{\color{accGreen}+ dialogue:} ``\,`That's very sad,' she {\color{accGreen}\textbf{said}}. `I'm sorry,' {\color{accGreen}\textbf{said}} the King. `Let's the King.' {\color{accGreen}\textbf{cried}}\ldots''\\[1mm]
      \qlabel{\color{accOrange}+ verse:} ``very that she was now the farmer\ldots\\ \textit{``You've been the day was getting so bad,''}\\ \ldots''
    \end{quotebox}
    \textbf{Features compose.} Poe + questions + dialogue + simplicity:
    \begin{quotebox}
      \textit{``She asked her mommy, `Can I go outside and play?' Her mommy said, `Yes, but you must stay on the swing.'\,''}
    \end{quotebox}

    \vspace{1mm}
    \begin{tcolorbox}[enhanced, arc=3mm,
      colback=highlightBg, colframe=highlightBg, boxrule=0pt,
      left=4mm, right=4mm, top=2mm, bottom=2mm,
      before skip=2mm, after skip=0mm]
      {\large\bfseries\color{accBlack}$\rightarrow$ Try it yourself!}\par\smallskip
      {\normalsize Pick an author, drag the simplicity slider, watch the voice change.\par\smallskip
      \textbf{\color{accPurple}\texttt{sixteen-voices.streamlit.app}}}
    \end{tcolorbox}
  \end{qpanel}

  % ─── Q7 DETECTION ≠ STEERING ───
  \begin{qpanel}[accOrange]
    \qtitle{Q7 $\cdot$ Does detection equal steering?}
    \textbf{No.} Archaic-pronoun features fire perfectly on ``thou'' in Blake and Milton. But injecting them produces nothing archaic --- the model collapses first.

    \textbf{Perfect detectors, useless steering vectors.} Steering amplifies only what the model can already express.
  \end{qpanel}

  % ─── Q8 CONCLUSION ───
  \begin{qpanel}[accRed]
    \qtitle{Q8 $\cdot$ Where does the style live in this model?}
    \colorbox{highlightBg}{\parbox{\dimexpr\linewidth-8mm}{\bfseries Style lives in two layers.}}

    \vspace{1mm}
    {\color{accBlue}\textbf{STRUCTURAL}} --- universal knobs (simplicity, dialogue, verse). Steer on \emph{any} adapter.

    {\color{accRed}\textbf{SEMANTIC}} --- author-specific fingerprints (dark atmosphere, cozy food, dialect). Detect everywhere. Steer only with the matching adapter.

    LoRAs \textbf{amplify}, they don't create. And the strongest style direction lives in the \textbf{\color{accOrange}MLP} --- invisible to any head.
  \end{qpanel}
